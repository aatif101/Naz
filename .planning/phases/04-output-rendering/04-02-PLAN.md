"""Naz CLI - Security scanning for solo developers."""

import typer
from rich.console import Console
from rich.panel import Panel

from naz.detection import (
    NodeNotFoundError,
    SpecfyError,
    SpecfyTimeoutError,
    run_specfy,
)

app = typer.Typer(
    name="naz",
    help="Security scanning for solo developers.",
    no_args_is_help=True,
)

console = Console(stderr=True)


@app.command()
def scan(
    path: str = typer.Argument(".", help="Path to the repository to scan"),
) -> None:
    """Scan a repository and detect its technology stack."""
    try:
        raw = run_specfy(path)
    except NodeNotFoundError:
        console.print(Panel(
            "Node.js is not installed.\n\n"
            "Install from [link=https://nodejs.org]https://nodejs.org[/link]"
            " then run: [bold]naz scan .[/bold]",
            title="[red]Error: Node.js Required[/red]",
            border_style="red",
        ))
        raise typer.Exit(code=1)
    except SpecfyTimeoutError:
        console.print(Panel(
            "Stack analysis timed out after 120 seconds.\n\n"
            "Try scanning a smaller directory.",
            title="[red]Error: Timeout[/red]",
            border_style="red",
        ))
        raise typer.Exit(code=1)
    except SpecfyError as exc:
        detail = exc.stderr[:500] if exc.stderr else str(exc)
        console.print(Panel(
            f"Stack analysis failed.\n\n{detail}",
            title="[red]Error: Analysis Failed[/red]",
            border_style="red",
        ))
        raise typer.Exit(code=1)

    # Phase 3 will replace this with ProjectProfile conversion
    typer.echo(raw)       # <-- THIS STUB IS REPLACED IN THIS TASK


@app.command()
def version() -> None:
    """Show the current version of naz."""
    from naz import __version__

    typer.echo(f"naz {__version__}")

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Specfy output → cli.py success path | Raw dict from run_specfy() flows into normalize() then render() or model_dump_json() |
| stdout → pipe consumer | When --json active, stdout must be clean JSON — no Rich markup contamination |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06 | Injection | `print(profile.model_dump_json(indent=2))` in cli.py success path | accept | Pydantic model_dump_json() serializes typed fields only. The `raw` field is excluded via `exclude=True`. No user-controlled keys can inject content into the JSON schema — the output shape is fixed by the Pydantic model definition. Low risk. |
| T-04-07 | Information Disclosure | `profile.model_dump_json()` — what fields are present? | accept | JSON contains: path (user-provided scan target), technologies, languages, dependencies, ai_dependencies, components. No credentials, tokens, API keys, or environment variables. This is intentional machine-readable output per OUT-02. |
| T-04-08 | Injection | `--json` flag parsed by Typer | accept | `typer.Option(False, "--json")` produces a Python bool. No string parsing, no shell injection surface. Boolean flag cannot carry injection payload. |
| T-04-09 | Spoofing | Malicious repo with crafted path that looks like a different path | accept | profile.path is printed in the header by renderer.py (handled in Plan 01 with markup.escape). cli.py does not print the path directly in the success path — delegated entirely to render(). |
</threat_model>
