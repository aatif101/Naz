class Dependency(BaseModel):
    manager: str        # e.g. "python", "npm"
    name: str           # e.g. "typer", "express"
    version: str | None = None

class Technology(BaseModel):
    name: str           # e.g. "python", "fastapi"
    category: str       # e.g. "runtime", "framework", "database", "infrastructure", "devtools", "ai_ml", "other"

class ProjectProfile(BaseModel):
    path: str                              # scanned directory path
    technologies: list[Technology]         # detected techs with categories
    languages: dict[str, int]             # language -> line count
    dependencies: list[Dependency]        # non-AI dependencies

    components: list[str]                 # sub-component names

    ```python
    """Naz renderer -- formats a ProjectProfile as Rich terminal panels."""

    from __future__ import annotations

    from rich.console import Console
    from rich.panel import Panel
    from rich.table import Table

    from naz.models import ProjectProfile

    # stdout Console — auto-detects TTY, strips ANSI when piped (D-12, D-13)
    # IMPORTANT: do NOT use Console(stderr=True) here — that is cli.py's error console
    _console = Console()


    def render(profile: ProjectProfile) -> None:
        """Render a ProjectProfile to the terminal as Rich panels.

        Section order (D-02): Header → Languages → Technologies → Dependencies → AI/LLM
        AI/LLM panel is omitted entirely when ai_dependencies is empty (D-07).
        Output goes to stdout so it can be piped and redirected (D-12).
        """
        # Header (D-03)
        _console.print(f"\n[bold]Scan complete:[/bold] {profile.path}\n")

        # Languages (D-04)
        _render_languages(profile)

        # Technologies (D-05)
        _render_technologies(profile)

        # Dependencies (D-06)
        _render_dependencies(profile)

        # AI/LLM — omit entirely if empty (D-07)
        if profile.ai_dependencies:
            _render_ai(profile)


    def _render_languages(profile: ProjectProfile) -> None:
        table = Table(show_header=True, header_style="bold")
        table.add_column("Language")
        table.add_column("Lines", justify="right")
        if profile.languages:
            for lang, count in profile.languages.items():
                table.add_row(lang, str(count))
        else:
            table.add_row("[dim]None detected[/dim]", "")
        _console.print(Panel(table, title="Languages", border_style="blue"))


    def _render_technologies(profile: ProjectProfile) -> None:
        table = Table(show_header=True, header_style="bold")
        table.add_column("Category")
        table.add_column("Technology")
        if profile.technologies:
            for tech in profile.technologies:
                table.add_row(tech.category, tech.name)
        else:
            table.add_row("[dim]None detected[/dim]", "")
        _console.print(Panel(table, title="Technologies", border_style="blue"))


    def _render_dependencies(profile: ProjectProfile) -> None:
        table = Table(show_header=True, header_style="bold")
        table.add_column("Manager")
        table.add_column("Package")
        table.add_column("Version")
        if profile.dependencies:
            for dep in profile.dependencies:
                table.add_row(dep.manager, dep.name, dep.version or "\u2014")
        else:
            table.add_row("[dim]None detected[/dim]", "", "")
        count = len(profile.dependencies)
        _console.print(Panel(table, title=f"Dependencies ({count})", border_style="blue"))


    def _render_ai(profile: ProjectProfile) -> None:
        table = Table(show_header=True, header_style="bold magenta")
        table.add_column("Manager")
        table.add_column("Package")
        table.add_column("Version")
        for dep in profile.ai_dependencies:
            table.add_row(dep.manager, dep.name, dep.version or "\u2014")
        count = len(profile.ai_dependencies)
        _console.print(Panel(
            table,
            title=f"[bold magenta]AI / LLM Components ({count})[/bold magenta]",
            border_style="yellow",
        ))
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Specfy output → renderer | Tech names and dependency names from external npx subprocess land in Rich markup strings |
| ProjectProfile fields → stdout | profile.path, tech names, dep names are printed to terminal |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Injection | `_console.print(f"Scan complete: {profile.path}")` | mitigate | profile.path is a string derived from Specfy's `path` list joined with `/`. It may contain Rich markup characters like `[` and `]`. Rich parses these as markup. Use `_console.print(f"Scan complete: ", profile.path, sep="")` or escape via `rich.markup.escape(profile.path)` to prevent a malicious path like `[bold red]evil[/bold red]` from altering terminal rendering. |
| T-04-02 | Injection | `table.add_row(lang, str(count))` — language names from Specfy | mitigate | Language names (e.g. "Python") are unlikely vectors but Specfy derives them from file analysis. Use `rich.markup.escape()` on all string values inserted into table rows: tech names, dep names, dep manager, language names. |
| T-04-03 | Injection | `table.add_row(tech.category, tech.name)` — tech names from Specfy | mitigate | Same as T-04-02. tech.name and tech.category must be escaped before passing to add_row(). |
| T-04-04 | Information Disclosure | `print(profile.model_dump_json(indent=2))` in Plan 02 | accept | JSON output contains only what Specfy detected (tech names, dep names, versions, path). No credentials, tokens, or secrets are in ProjectProfile. The `raw` field is excluded via `exclude=True`. Risk is low — this is intentional machine-readable output. |
| T-04-05 | Denial of Service | Extremely large dependency lists (thousands of entries) | accept | Rich Table can handle thousands of rows. The CLI is point-in-time, not a long-running service. Specfy scans real repos; pathological input would require a maliciously crafted repo. Accept for v0.1. |

**ASVS L1 Actions Required (blocking):**
- T-04-01 and T-04-02 and T-04-03: Apply `rich.markup.escape()` to all user-derived string values passed to Rich console and table rows. This is a one-line addition per field and must be implemented in Task 2.
</threat_model>
